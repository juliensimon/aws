CREATE EXTERNAL TABLE IF NOT EXISTS athenatest.salesparquet (
  `lastname` string,
  `firstname` string,
  `gender` string,
  `state` string,
  `age` int,
  `day` int,
  `hour` int,
  `minutes` int,
  `items` int,
  `basket` int 
)
ROW FORMAT SERDE 'org.apache.hadoop.hive.ql.io.parquet.serde.ParquetHiveSerDe'
WITH SERDEPROPERTIES (
  'serialization.format' = '1'
) LOCATION 's3://jsimon-redshift-demo-us/dataparquet/';
